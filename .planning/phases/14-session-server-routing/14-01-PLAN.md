```typescript
import { readFileSync, writeFileSync, mkdirSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { homedir } from 'node:os';

export interface SessionEntry {
  server: string;  // name from registry (must match a ServerEntry.name in servers.json)
  url: string;     // full http://host:port URL — stored alongside name so error messages show both without re-lookup
}

export interface SessionMap {
  sessions: Record<string, SessionEntry>;
}

const SESSIONS_DIR = join(homedir(), '.config', 'prefect');
export const SESSIONS_PATH = join(SESSIONS_DIR, 'sessions.json');

export function readSessionMap(sessionsPath: string = SESSIONS_PATH): SessionMap {
  try {
    const parsed = JSON.parse(readFileSync(sessionsPath, 'utf8'));
    if (!parsed || typeof parsed.sessions !== 'object' || Array.isArray(parsed.sessions)) {
      throw new Error(`malformed sessions map at ${sessionsPath}: expected { sessions: { ... } }`);
    }
    return parsed as SessionMap;
  } catch (err) {
    const code = (err as NodeJS.ErrnoException).code;
    if (code === 'ENOENT') return { sessions: {} };
    throw new Error(`could not parse ${sessionsPath}: ${(err as Error).message}`);
  }
}

export function writeSessionMap(map: SessionMap, sessionsPath: string = SESSIONS_PATH): void {
  mkdirSync(dirname(sessionsPath), { recursive: true });
  writeFileSync(sessionsPath, JSON.stringify(map, null, 2) + '\n');
}

export function addSession(sessionId: string, entry: SessionEntry, sessionsPath: string = SESSIONS_PATH): void {
  const map = readSessionMap(sessionsPath);
  map.sessions[sessionId] = entry;
  writeSessionMap(map, sessionsPath);
}

export function removeSession(sessionId: string, sessionsPath: string = SESSIONS_PATH): void {
  const map = readSessionMap(sessionsPath);
  if (!(sessionId in map.sessions)) return;  // silent no-op (D-12 cleanup must be idempotent)
  delete map.sessions[sessionId];
  writeSessionMap(map, sessionsPath);
}

export function lookupSession(sessionId: string, sessionsPath: string = SESSIONS_PATH): SessionEntry | undefined {
  return readSessionMap(sessionsPath).sessions[sessionId];
}
```

Key differences from registry.ts (intentional):
- Interface `SessionEntry` has `{ server, url }` (not `{ name, host, port, model }`)
- ENOENT fallback returns `{ sessions: {} }` (object) not `{ servers: [] }` (array)
- Malformed check uses `typeof parsed.sessions !== 'object' || Array.isArray(parsed.sessions)` (registry uses `!Array.isArray(parsed.servers)`) — sessions is a Record, not an Array
- `removeSession` is a SILENT no-op when sessionId is unknown (returns early without writing) — registry's `removeServer` throws. This difference is intentional per D-12: stale-session cleanup must be idempotent; calling removeSession on an already-removed id during a 404 retry must not throw.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP tool input → sessions.json | Each tool call writes a sessionId (from OpenCode) plus a server URL (resolved from registry or env var) to a user-owned config file |
| sessions.json file → all subsequent tool calls | A malicious or corrupted sessions.json could redirect every sessionId-bearing tool call to an attacker-controlled URL |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-14-01 | Tampering | sessions.json on disk | accept | Personal-use localhost service. File lives in user-only `~/.config/prefect/` (default 0700 dir from mkdirSync). No multi-user threat in scope per REQUIREMENTS.md "Out of Scope" → "Multi-user / team concerns". |
| T-14-02 | Information Disclosure | sessions.json contents (sessionIds + server URLs) | accept | sessionIds are opaque ULIDs, not secrets; server URLs already exist in plaintext servers.json. No new disclosure vector introduced. |
| T-14-03 | Spoofing | Malformed sessions.json crafted to redirect tool calls | mitigate | `readSessionMap` validates `typeof parsed.sessions === 'object' && !Array.isArray(parsed.sessions)` and throws on malformed shape (test 8). Tool call cannot silently route to a wrong server because lookup throws on bad file. |
| T-14-04 | Denial of Service | Unbounded sessionId growth in sessions.json | accept | Stale-session cleanup (Plan 03) prunes 404s; map size bounded by active sessions on registered servers. No automatic GC needed for personal-use scale. |
</threat_model>
